\newcommand{\plotthat}[4]{%
		\addplot [color=#1, thick]
			table [
				col sep=comma,
				trim cells=true,
				y=#2,
			    x=#3
				] 
				{#4};
}
\newcommand{\plotmark}[6]{%
		\addplot [color=#1,mark=#5]
			table [
				col sep=comma,
				trim cells=true,
				y=#2,
			           x=#3,
			           each nth point={#6},
				] 
				{#4};
}

\newcommand{\pplot}[2]{%
		\addplot [color=#1, thick]
			table [
				col sep=comma,
				trim cells=true,
				y=Y,
			    	x=X
				] 
				{#2};
}
\newcommand{\effplot}[3]{
	\addplot [color=black, 
			thick,
			dotted,
			mark size=0.1pt,
			#3]
			table [
			col sep=comma,
			trim cells=true,
			y=head,
		    	x=volflow
			] 
		{#2};
}
